def hello():
    print("Hello, world!")

def greet(name, title=""):
    """Greet someone with an optional title."""
    if title:
        return f"Hello, {title} {name}!"
    return f"Hello, {name}!"

class Calculator:
    def add(self, a, b):
        return a + b
    
    def subtract(self, a, b):
        return a - b

if __name__ == "__main__":
    hello()
    calc = Calculator()
    print(calc.add(1, 2))
